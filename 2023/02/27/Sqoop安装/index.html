# Sqoop 安装

# 一、解压

tar -zxvf /opt/software/sqoop-1.4.2.bin__hadoop-2.0.0-alpha.tar.gz -C /opt/module

# 二、配置 profile 文件

vi /etc/profile

#sqoop
export SQOOP_HOME=/opt/module/sqoop-1.4.2.bin__hadoop-2.0.0-alpha
export PATH=$PATH:$ZOOKEEPER_HOME/bin:$SQOOP_HOME/bin

source /etc/profile  (刷新profile文件)

验证环境是否配置完成

sqoop version

# 三、修改 sqoop-env.sh 文件

cd /opt/module/sqoop-1.4.2.bin__hadoop-2.0.0-alpha/conf

mv sqoop-env-template.sh sqoop-env.sh
vi sqoop-env.sh

#Set path to where bin/hadoop is available
export HADOOP_COMMON_HOME=/opt/module/hadoop-2.7.7

#Set path to where hadoop-*-core.jar is available
export HADOOP_MAPRED_HOME=/opt/module/hadoop-2.7.7

#set the path to where bin/hbase is available
export HBASE_HOME=

#Set the path to where bin/hive is available
export HIVE_HOME=/opt/module/apache-hive-2.3.4-bin

#Set the path for where zookeper config dir is
export ZOOCFGDIR=/opt/module/zookeeper-3.4.6/conf

# 将 mysql 关系型数据库驱动包放到 sqoop/lib 目录下

cp /opt/software/mysql-connector-java-5.1.47-bin.jar /opt/module/sqoop-1.4.2.bin__hadoop-2.0.0-alpha/lib

# 验证 sqoop 连接 mysql 数据库是否成功

sqoop list-databases --connect jdbc:mysql://master:3306 --username root --password 123456
