# hive 和 mysql 安装

# 一、解压（# 注意：Hive 只需要在 master 节点上安装配置）

cd  /opt/software
tar -zxvf apache-hive-2.3.4-bin.tar.gz -C /opt/module

# 二、安装 mysql

先卸载 mysql

	查看本机	rpm -qa | grep mariadb

	卸载	rpm -e --nodeps XXXXXXXXXX

安装

tar -xvf mysql-5.7.20-1.el6.x86_64.rpm-bundle.tar -C /opt/module

cd /opt/module

rpm -ivh mysql-community-common-5.7.20-1.el6.x86_64.rpm

rpm -ivh mysql-community-libs-5.7.20-1.el6.x86_64.rpm

rpm -ivh mysql-community-client-5.7.20-1.el6.x86_64.rpm

rpm -ivh mysql-community-server-5.7.20-1.el6.x86_64.rpm --force --nodeps

* 检查一下 *

rpm -qa | grep mysql

* 启动服务 *

systemctl daemon-reload	#重载所有修改过的配置文件

systemctl start mysqld	#开启服务

systemctl enable mysqld #开机自启 若不成功也不影响

* 登录 MySQL 数据库 *

grep "temporary password" /var/log/mysqld.log	 #获取初密码

mysql -uroot -p	#登陆MySQL（注意中英文）

*MySQL 密码安全策略设置 *

set global validate_password_policy=0; #设置密码强度为低级

set global validate_password_length=4;	#设置密码长度

alter user 'root'@'localhost' identified by '123456';	#修改用户密码

\q	#退出

* 设置远程登录 *

mysql -uroot -p123456	#以新密码登陆MySQL

create user 'root'@'%' identified by '123456';	#创建用户

grant all privileges on *.* to 'root'@'%' with grant option;	#***\*允许远程连接\****

flush privileges;	#刷新权限

create database if not exists test;		#创建数据库test

\q

# 第二种方式

解压并改名

tar -zxvf /opt/software/mysql-5.7.30-linux-glibc2.12-x86_64.tar.gz -C /opt/module

mv mysql-5.7.30-linux-glibc2.12-x86_64 mysql

创建数据目录并赋予权限

mkdir -p  /data/mysql              #创建目录
chown mysql:mysql -R /data/mysql   #赋予权限

配置 my.cnf (新文件)

vi /etc/my.cnf

[mysqld]
bind-address=0.0.0.0
port=3306
user=mysql
basedir=/opt/module/mysql
datadir=/data/mysql
socket=/tmp/mysql.sock
log-error=/data/mysql/mysql.err
pid-file=/data/mysql/mysql.pid
#character config
character_set_server=utf8mb4
symbolic-links=0
explicit_defaults_for_timestamp=true

进入 MySQL 的 bin 目录

cd /opt/module/mysql/bin

初始化

./mysqld --defaults-file=/etc/my.cnf --basedir=/opt/module/mysql/ --datadir=/data/mysql/ --user=mysql --initialize

查看密码

cat /data/mysql/mysql.err

先将 mysql.server 放置到 /etc/init.d/mysql 中

cp /usr/local/mysql/support-files/mysql.server /etc/init.d/mysql

启动

service mysql start

ps -ef|grep mysql

./mysql -u root -p   #bin目录下

再执行

SET PASSWORD = PASSWORD('123456');
ALTER USER 'root'@'localhost' PASSWORD EXPIRE NEVER;
FLUSH PRIVILEGES;  
\q

再登陆

use mysql                                            #访问mysql库
update user set host = '%' where user = 'root';      #使root能再任何host访问
FLUSH PRIVILEGES;                                    #刷新

# 三、修改配置文件

配置环境文件

# vi /etc/profile

#set hive
export HIVE_HOME=/opt/module/apache-hive-2.3.4-bin
export PATH=$PATH:$HIVE_HOME/bin

source  /etc/profile

# 将 mysql 依赖包放入 master 中

# 修改文件

mkdir /opt/module/apache-hive-2.3.4-bin/warehouse

cd /opt/module/apache-hive-2.3.4-bin/conf
# vi hive-site.xml
<configuration>
<!-- Hive产生的元数据存放位置-->
<property>
<name>hive.metastore.warehouse.dir</name>
<value>/opt/module/apache-hive-2.3.4-bin/warehouse</value>
</property>
<!-- 数据库连接JDBC的URL地址-->
<property>
<name>javax.jdo.option.ConnectionURL</name> 
<value>jdbc:mysql://master:3306/hive?createDatabaseIfNotExist=true&amp;useSSL=false</value>
</property>
<!-- 数据库连接driver，即MySQL驱动-->
<property>
<name>javax.jdo.option.ConnectionDriverName</name>
<value>com.mysql.jdbc.Driver</value>
</property>
<!-- MySQL数据库用户名-->
<property>
<name>javax.jdo.option.ConnectionUserName</name>
<value>root</value>
</property>
<!-- MySQL数据库密码-->
<property>
<name>javax.jdo.option.ConnectionPassword</name>
<value>123456</value>
</property>
<property>
<name>hive.metastore.schema.verification</name>
<value>false</value>
</property>
<property>
<name>datanucleus.schema.autoCreateAll</name>
<value>true</value>
</property>
</configuration>

# 四、修改环境变量

cd /opt/module/apache-hive-2.3.4-bin/conf

mv hive-env.sh.template hive-env.sh

# vi hive-env.sh

HADOOP_HOME=/opt/module/hadoop-2.7.7
export HIVE_CONF_DIR=/opt/module/apache-hive-2.3.4-bin/conf

schematool -initSchema -dbType mysql	初始化mysql

# 五、启动 Mysql 和 Hive

# 启动 hadoop 集群（只 master 上）

cd /opt/module/hadoop-2.7.7
sbin/start-all.sh

# 元数据初始化命令

schematool -dbType mysql -initSchema

# 启动 hive client (只 master 上)

cd /opt/module/apache-hive-2.3.4-bin
bin/hive --service metastore &
bin/hive

将 MySQL 驱动复制到 java 的 jre 下的 lib 下的 ext 目录下

# 六、测试 hive 是否启动成功

# 查看数据库

hive>show databases;

# 创建数据库

hive>create database testbase;

# 复制一个 master 会话，查看一下 master 进程

如果可以查看以及建立数据库，进程中有RunJar进程，则代表Mysql和Hive安装成功啦~

# 解决版本冲突问题 (slave1 操作)

cp /opt/module/apache-hive-2.3.4-bin/lib/jline-2.12.jar /opt/module/hadoop-2.7.7/share/hadoop/yarn/lib
