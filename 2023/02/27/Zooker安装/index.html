# zooker 安装

# 一、解压

tar -zxvf zookeeper-3.4.6.tar.gz -C /opt/module

# 二、修改配置文件

cd /opt/module/zookeeper-3.4.6
mkdir data
cd data

# vi myid 机器的 ID

填入 1

cd /opt/module/zookeeper-3.4.6/conf

mv zoo_sample.cfg zoo.cfg	改名

# vi zoo.cfg 添加 datadir 的路径和添加 server 的配置

dataDir=/opt/module/zookeeper-3.4.6/data
server.1=master:2888:3888
server.2=slave1:2888:3888
server.3=slave2:2888:3888

# 三、配置 Zookeeper 环境变量

# vi /etc/profile

#set zookeeper environment  
export ZOOKEEPER_HOME=/opt/module/zookeeper-3.4.6 
export PATH=$PATH:$ZOOKEEPER_HOME/bin 

source /etc/profile		#生效环境变量

# 四、同步到其它虚拟机

scp -r /opt/module/zookeeper-3.4.6 root@slave1:/opt/module 

scp -r /opt/module/zookeeper-3.4.6 root@slave2:/opt/module

scp -r /etc/profile root@slave1:/etc/profile

scp -r /etc/profile root@slave2:/etc/profile

source /etc/profile		#slave1和slave2都要执行

# 五、修改分节点的 myid 文件

cd /opt/module/zookeeper-3.4.6/data

# vi myid 机器 ID

分别为2和3

# 六、启动 Zookeeper（每个节点都要进行）

cd /opt/module/zookeeper-3.4.6/bin

zkServer.sh start		#开启服务

zkServer.sh status		#查看状态

注：一个节点是 Leader，其余的结点是 Follower（Leader 不一定是主节点，系统随机分配的）。
