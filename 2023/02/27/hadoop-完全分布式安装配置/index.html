# 完全分布式安装 hadoop

# 1、新建 /opt/software 使用 xftp 将 hadoop 和 java 的压缩包上传

cd /opt/
mkdir software

# 2、新建 module 文件夹存放解压后的 hadoop

mkdir module

# 3、解压 hadoop

cd /opt/software
tar -zxf hadoop-2.7.7.tar.gz -C /opt/module/

# 4、配置环境

# vi /etc/profile

#hadoop
export HADOOP_HOME=/opt/module/hadoop-2.7.7
export PATH=$PATH:$HADOOP_HOME/bin
export PATH=$PATH:$HADOOP_HOME/sbin

# 5、使其生效

source /etc/profile

# 6、验证

hadoop version

# 7、配置 hadoop 文件

# vi core-site.xml

<configuration>
<!-- 指定HDFS中NameNode的地址 -->
<property>
 <name>fs.defaultFS</name>
 <value>hdfs://master:9000</value>
</property>
 <!-- 指定hadoop运行时产生文件的存储目录 -->
<property>
 <name>hadoop.tmp.dir</name>
 <value>/opt/module/hadoop-2.7.7/tmp</value>
</property>
</configuration>

# vi hdfs-site.xml

<configuration>
 <property>
	<name>dfs.replication</name>
  	<value>3</value>
 </property>
 <property>
  	<name>dfs.namenode.secondary.http-address</name>
	<value>slave1:50090</value>
 </property>
 <property>
	<name>dfs.data.dir</name>
	<value>/opt/module/hadoop-2.7.7/tmp/data</value>
 </property>
 <property>
	<name>dfs.name.dir</name>
	<value>/opt/module/hadoop-2.7.7/tmp/name</value>
 </property>
</configuration>

* 配置文件 slaves slaves 文件里面记录的是集群里所有 DataNode 的主机名 *

# vi slaves

master
slave1
slave2

# vi yarn-site.xml

<configuration>
<!-- Site specific YARN configuration properties -->
<!-- reducer获取数据的方式 -->
<property>
 <name>yarn.nodemanager.aux-services</name>
 <value>mapreduce_shuffle</value>
</property>
<!-- 指定YARN的ResourceManager的地址 -->
<property>
<name>yarn.resourcemanager.hostname</name>
<value>master</value>
 </property>
 <!-- 关闭虚拟内存的检测 -->
 <property>  
    <name>yarn.nodemanager.vmem-check-enabled</name>
    <value>false</value>
</property> 
</configuration>

# vi yarn-env.sh

export JAVA_HOME=/opt/module/jdk1.8

# vi hadoop-env.sh

export JAVA_HOME=/opt/module/jdk1.8

# vi mapred-site.xml

先改名，因为本身是没有mapred-site.xml这个文件的

mv mapred-site.xml.template mapred-site.xml

<configuration>
<!-- 指定mr运行在yarn上 -->
 <property>
	<name>mapreduce.framework.name</name>
	<value>yarn</value>
 </property>
 <property>
	<name>mapreduce.jobhistory.address</name>
	<value>master:10020</value>
 </property>
 <property>
	<name>mapreduce.jobhistory.webapp.address</name>
	<value>master:19888</value>
 </property>
</configuration>

# 8、把 master 下修改的文件分发到 slave1 和 slave2 下

scp -r /opt/module/ root@slave1:/opt

scp -r /opt/module/ root@slave2:/opt

scp -r /etc/profile root@slave1:/etc/profile

scp -r /etc/profile root@slave2:/etc/profile

source /etc/profile	#分别在两个分节点上

# 9、测试集群

hdfs namenode -format	先格式化（只在主节点上）
# 启动
start-all.sh	#只在主节点上

jps
# 打开浏览器 http://master:50070/
